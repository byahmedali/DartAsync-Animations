Dart Main Isolate

import 'dart:async';
import 'dart:io';
import 'dart:isolate';


String fetchData(int param) {
    print("Do something with $param...");
    sleep(Duration(seconds: param));
    print("Done with $param");
    return "Result of $param";
}
  
  
Future<List<String>> mainLogic() async {
    // Run in Isolate (similar to Thread/Process)
    var task1 = Isolate.run(() => fetchData(1));
    var task2 = Isolate.run(() => fetchData(2));
    var result1 = await task1;
    print("Isolate 1 fully completed");
    var result2 = await task2;
    print("Isolate 2 fully completed");

    // Dart usually relies on isolates instead of OS processes
    // for true multiprocessing since it doesn't share memory.
    
    // (Additional execution could go here)
    
    return [result1, result2];
}


void main() async {
    var results = await mainLogic();
    print(results);
}
